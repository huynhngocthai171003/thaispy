Thư viện os trong Python Làm việc với hệ điều hành: file, thư mục, biến môi trường

Thư viện os trong Python dùng để làm việc với hệ điều hành, cho phép quản lý file, thư mục, biến môi trường, và nhiều thao tác khác.

I. Quản lý thư mục với os

Hành động Hàm Mô tả Thích hợp GUI
Tạo thư mục os.mkdir(path) Tạo một thư mục mới tại đường dẫn chỉ định Dùng guizero: Tạo 1 thư mục 
from guizero import App, TextBox, PushButton, Text, info, error
import os

def create_folder():
    folder_name = folder_input.value.strip()
    if folder_name == "":
        error("Lỗi", "Tên thư mục không được để trống.")
        return
    try:
        os.mkdir(folder_name)
        info("Thành công", f"Đã tạo thư mục: {folder_name}")
    except FileExistsError:
        error("Lỗi", "Thư mục đã tồn tại.")
    except Exception as e:
        error("Lỗi khác", str(e))

app = App("Tạo thư mục mới", width=400, height=200)
Text(app, text="Nhập tên thư mục:")
folder_input = TextBox(app, width=30)
PushButton(app, text="Tạo thư mục", command=create_folder)
app.display()

from guizero import App, TextBox, PushButton, Text, info, error
import os

def change_directory():
    folder = folder_input.value.strip()
    if folder == "":
        error("Lỗi", "Vui lòng nhập đường dẫn.")
        return
    try:
        os.chdir(folder)
        current_path.value = f"Đã chuyển đến: {os.getcwd()}"
        info("Thành công", f"Đang làm việc tại: {os.getcwd()}")
    except FileNotFoundError:
        error("Lỗi", "Thư mục không tồn tại.")
    except NotADirectoryError:
        error("Lỗi", "Đường dẫn không phải thư mục.")
    except Exception as e:
        error("Lỗi khác", str(e))

app = App("Chuyển thư mục làm việc", width=450, height=220)
Text(app, text="Nhập đường dẫn:")
folder_input = TextBox(app, width=40)
PushButton(app, text="Chuyển", command=change_directory)
current_path = Text(app, text=f"Thư mục hiện tại: {os.getcwd()}")
app.display()

from guizero import App, TextBox, PushButton, Text, ListBox, error

def list_files():
    path = folder_input.value.strip()
    if not path:
        path = "."
    try:
        items = os.listdir(path)
        file_list.clear()

        error("Lỗi", "Thư mục không tồn tại.")
    except NotADirectoryError:
        error("Lỗi", "Đường dẫn không phải thư mục.")
    except Exception as e:
        error("Lỗi khác", str(e))

Text(app, text="Nhập đường dẫn thư mục:")

Xóa tệp os.remove(path) Xóa một tệp Nhập đường dẫn → Xóa 
from guizero import App, TextBox, PushButton, Text, info, error
import os

def delete_file():
    file_path = input_box.value.strip()
    if not file_path:
        error("Lỗi", "Vui lòng nhập đường dẫn tệp cần xóa.")
        return
    try:
        os.remove(file_path)
        info("Thành công", f"Đã xóa tệp: {file_path}")
    except FileNotFoundError:
        error("Lỗi", "Tệp không tồn tại.")
    except IsADirectoryError:
        error("Lỗi", "Đây là thư mục, không phải tệp.")
    except PermissionError:
        error("Lỗi", "Không có quyền xóa tệp này.")
    except Exception as e:
        error("Lỗi khác", str(e))

app = App("Xóa tệp", width=500, height=200)
Text(app, text="Nhập đường dẫn tệp cần xóa:")
input_box = TextBox(app, width=50)
PushButton(app, text="Xóa tệp", command=delete_file)
app.display()
Đổi tên os.rename(src, dst) Đổi tên tệp hoặc thư mục Nhập tên cũ và mới 
from guizero import App, Text, TextBox, PushButton, info, error
import os

def rename_file_or_folder():
    old_name = old_input.value.strip()
    new_name = new_input.value.strip()

    if not old_name or not new_name:
        error("Lỗi", "Vui lòng nhập tên cũ và mới.")
        return

    try:
        os.rename(old_name, new_name)
        info("Thành công", f"Đã đổi tên: {old_name} → {new_name}")
    except FileNotFoundError:
        error("Lỗi", "Tệp hoặc thư mục không tồn tại.")
    except FileExistsError:
        error("Lỗi", "Tên mới đã tồn tại.")
    except Exception as e:
        error("Lỗi khác", str(e))

app = App("Đổi tên", width=500, height=220)
Text(app, text="Nhập tên cũ:")
old_input = TextBox(app, width=50)
Text(app, text="Nhập tên mới:")
new_input = TextBox(app, width=50)
PushButton(app, text="Đổi tên", command=rename_file_or_folder)
app.display()
